use db_members_info_seokwon;

CREATE TABLE `MEMBER_DEFAULT` (
    `MEMBER_DEFAULT_ID` BIGINT NOT NULL,
    `NAME` VARCHAR(255) NULL,
    `AGE` INT NULL,
    `GENDER` VARCHAR(255) NULL,
    `PHONE_NUM` VARCHAR(255) NULL,
    `EMAIL` VARCHAR(255) NULL,
    PRIMARY KEY (`MEMBER_DEFAULT_ID`)
);

CREATE TABLE `HOBBY` (
    `HOBBY_ID` BIGINT NOT NULL,
    `HOBBY_NAME` VARCHAR(255) NULL,
    PRIMARY KEY (`HOBBY_ID`)
);

CREATE TABLE `ADDRESS` (
    `ADDRESS_ID` BIGINT NOT NULL,
    `ADDRESS_NAME` VARCHAR(255) NULL,
    PRIMARY KEY (`ADDRESS_ID`)
);

CREATE TABLE `MEMBER_HOBBY` (
    `MEMBER_HOBBY_ID` BIGINT NOT NULL,
    `MEMBER_DEFAULT_ID` BIGINT NOT NULL,
    `HOBBY_ID` BIGINT NOT NULL,
    PRIMARY KEY (`MEMBER_HOBBY_ID`),
    FOREIGN KEY (`MEMBER_DEFAULT_ID`) REFERENCES `MEMBER_DEFAULT`(`MEMBER_DEFAULT_ID`),
    FOREIGN KEY (`HOBBY_ID`) REFERENCES `HOBBY`(`HOBBY_ID`)
);

CREATE TABLE `MEMBER_ADDRESS` (
    `MEMBER_ADDRESS_ID` BIGINT NOT NULL,
    `MEMBER_DEFAULT_ID` BIGINT NOT NULL,
    `ADDRESS_ID` BIGINT NOT NULL,
    PRIMARY KEY (`MEMBER_ADDRESS_ID`),
    FOREIGN KEY (`MEMBER_DEFAULT_ID`) REFERENCES `MEMBER_DEFAULT`(`MEMBER_DEFAULT_ID`),
    FOREIGN KEY (`ADDRESS_ID`) REFERENCES `ADDRESS`(`ADDRESS_ID`)
);

commit;
