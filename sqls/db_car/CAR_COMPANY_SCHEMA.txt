use db_car;

CREATE TABLE `CAR_DATAIL` (
	`CAR_DETAIL_ID`	VARCHAR(255)	NOT NULL,
	`CAR_ID`	BIGINT	NOT NULL,
	`COMPANY_ID`	BIGINT	NOT NULL,
	`CREATE_DATE`	DATE	NULL
);

CREATE TABLE `CAR_COMPANY` (
	`COMPANY_ID`	BIGINT	NOT NULL,
	`NAME`	VARCHAR(255)	NULL
);

CREATE TABLE `CAR_NAME` (
	`CAR_ID`	BIGINT	NOT NULL,
	`CAR_NAME`	VARCHAR(255)	NULL
);

CREATE TABLE `CAR_OPTION` (
	`OPTION_ID`	BIGINT	NOT NULL,
	`OPTION_NAME`	VARCHAR(255)	NULL
);

CREATE TABLE `CAR_OPTION_DETAIL` (
	`CAR_OPTION_DETAIL_ID`	BIGINT	NOT NULL,
	`CAR_ID`	BIGINT	NOT NULL,
	`OPTION_ID`	BIGINT	NOT NULL
);

ALTER TABLE `CAR_DATAIL` ADD CONSTRAINT `PK_CAR_DATAIL` PRIMARY KEY (
	`CAR_DETAIL_ID`
);

ALTER TABLE `CAR_COMPANY` ADD CONSTRAINT `PK_CAR_COMPANY` PRIMARY KEY (
	`COMPANY_ID`
);

ALTER TABLE `CAR_NAME` ADD CONSTRAINT `PK_CAR_NAME` PRIMARY KEY (
	`CAR_ID`
);

ALTER TABLE `CAR_OPTION` ADD CONSTRAINT `PK_CAR_OPTION` PRIMARY KEY (
	`OPTION_ID`
);

ALTER TABLE `CAR_OPTION_DETAIL` ADD CONSTRAINT `PK_CAR_OPTION_DETAIL` PRIMARY KEY (
	`CAR_OPTION_DETAIL_ID`
);
