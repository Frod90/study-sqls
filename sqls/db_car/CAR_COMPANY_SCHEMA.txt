use db_car;

CREATE TABLE `CAR_DETAIL` (
	`CAR_DETAIL_ID` VARCHAR(255) NOT NULL,
	`CAR_ID` BIGINT NOT NULL,
	`CREATE_DATE` DATE NULL,
	PRIMARY KEY (`CAR_DETAIL_ID`)
);

CREATE TABLE `CAR_COMPANY` (
	`COMPANY_ID` BIGINT NOT NULL,
	`NAME` VARCHAR(255) NULL,
	PRIMARY KEY (`COMPANY_ID`)
);

CREATE TABLE `CAR_NAME` (
	`CAR_ID` BIGINT NOT NULL,
	`CAR_NAME` VARCHAR(255) NULL,
	`COMPANY_ID` BIGINT NOT NULL,
	PRIMARY KEY (`CAR_ID`),
	FOREIGN KEY (`COMPANY_ID`) REFERENCES `CAR_COMPANY`(`COMPANY_ID`)
);

CREATE TABLE `CAR_OPTION` (
	`OPTION_ID` BIGINT NOT NULL,
	`OPTION_NAME` VARCHAR(255) NULL,
	PRIMARY KEY (`OPTION_ID`)
);

CREATE TABLE `CAR_OPTION_DETAIL` (
	`CAR_OPTION_DETAIL_ID` BIGINT NOT NULL,
	`CAR_ID` BIGINT NOT NULL,
	`OPTION_ID` BIGINT NOT NULL,
	PRIMARY KEY (`CAR_OPTION_DETAIL_ID`),
	FOREIGN KEY (`CAR_ID`) REFERENCES `CAR_NAME`(`CAR_ID`),
	FOREIGN KEY (`OPTION_ID`) REFERENCES `CAR_OPTION`(`OPTION_ID`)
);
